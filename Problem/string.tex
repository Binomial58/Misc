\noindent
\textbf{問題}

\medskip
初期関数を $f_0(x)=x^{12}$ とする．

\medskip
次の 4 つの操作を定める．
\begin{equation*}
\begin{aligned}
D(f)(x)&=\frac{d}{dx}f(x),\quad
I(f)(x)=\int_0^x f(t)\,dt,\quad
L(f)(x)=f(2x),\quad
R(f)(x)=f\!\left(\frac{x}{2}\right).
\end{aligned}
\end{equation*}

\medskip
文字列
\begin{equation*}
T=t_1t_2\cdots t_M \qquad (t_i\in\{D,I,L,R\})
\end{equation*}
を考える．
$f_T$ を「$f_0$ に $T$ を左から順に適用した結果」とする．
例えば
\begin{equation*}
T=DL \Longrightarrow f_{DL}=L(D(f_0)).
\end{equation*}

\medskip
記号の意味は次のとおり．
\begin{itemize}
  \item $D^a$ ：$D$ を $a$ 回適用する．
  \item $(U)^q$ ：操作列 $U$ を $q$ 回連結する．
\end{itemize}

\medskip
次の操作列 $S$ を考える．
\begin{equation*}
S=
\bigl(D^4L^7I^4R^7\bigr)^{97}
\bigl(I^3R^5D^3L^5\bigr)^{103}
L^2D^5R
\end{equation*}

$S$ を $f_0$ に左から順に適用して得る関数を $f_S$ とする．
最終形は単項式で，
\begin{equation*}
f_S(x)=\frac{A}{2^B}x^K
\end{equation*}
と書けることが保証される．
$A$ を奇数として $A,B,K$ を求めよ．

\medskip
\textbf{答え}
\begin{equation*}
f_S(x)=\frac{1485}{2^{4238}}x^7
\end{equation*}
したがって
\begin{equation*}
A=1485,\qquad B=4238,\qquad K=7
\end{equation*}
である．
